3.ODS层：用户行为原始数据表（timestamp 是 Hive 保留关键字，作为列名会导致解析错误；改用 event_ts 表达“事件时间戳”。Hive 不支持 DATETIME 数据类型，改用 TIMESTAMP 类型，列名改为 event_time 避免歧义。）
DROP TABLE IF EXISTS ecommerce.ods_user_behavior;

CREATE TABLE ecommerce.ods_user_behavior(
    user_id      INT          COMMENT '用户ID',
    item_id      INT          COMMENT '商品ID',
    brand        VARCHAR(100) COMMENT '品牌',
    brand_id     INT          COMMENT '品牌ID',
    item_name    VARCHAR(255) COMMENT '商品名称',
    category     VARCHAR(100) COMMENT '商品类别',
    category_id  INT          COMMENT '商品类目ID',
    behavior     VARCHAR(20)  COMMENT '行为类型',
    event_ts     BIGINT       COMMENT 'Unix时间戳（毫秒）',
    price        DECIMAL(10,2) COMMENT '商品售价',
    event_time   TIMESTAMP    COMMENT '行为时间',
    price_level  VARCHAR(20)  COMMENT '价格区间'
)
ROW FORMAT DELIMITED
FIELDS TERMINATED BY ','      -- 明确指定逗号分隔
STORED AS TEXTFILE
TBLPROPERTIES ("skip.header.line.count"="1");   -- 如果文件有表头则保留，否则删除

7.建立 DWS（数据汇总层）
(1)品牌销售统计（dws_brand_sales）统计每个品牌：总销量（buy次数）总销售额（price求和）
CREATE TABLE dws_brand_sales(
    brand_id INT ,
    brand VARCHAR(100),
    buy_count INT,
    sales_amount DECIMAL(12,2)

);

INSERT INTO dws_brand_sales
SELECT
    g.brand_id,
    b.brand,
    COUNT(*) AS buy_count,
    ROUND(SUM(f.price),2) AS sales_amount
FROM fact_behavior f
JOIN dim_goods g
ON f.item_id = g.item_id
JOIN dim_brand b
ON g.brand_id = b.brand_id
WHERE f.behavior='buy'
GROUP BY g.brand_id,b.brand;

SELECT *
FROM dws_brand_sales
ORDER BY sales_amount DESC
LIMIT 10;

(2)热门商品统计（dws_item_hot）统计：浏览量(pv)收藏数(fav)购买量(buy)
DROP TABLE IF EXISTS dws_item_hot;

CREATE TABLE dws_item_hot(
    item_id INT ,
    item_name VARCHAR(255),
    pv_count INT,
    fav_count INT,
    
    buy_count INT

);
INSERT INTO dws_item_hot
SELECT
    g.item_id,
    g.item_name,
    SUM(CASE WHEN f.behavior='pv' THEN 1 ELSE 0 END) AS pv_count,
SUM(CASE WHEN f.behavior='fav' THEN 1 ELSE 0 END) AS fav_count,

SUM(CASE WHEN f.behavior='buy' THEN 1 ELSE 0 END) AS buy_count
FROM fact_behavior f
JOIN dim_goods g
ON f.item_id=g.item_id
GROUP BY g.item_id,g.item_name;

SELECT *
FROM dws_item_hot
ORDER BY buy_count DESC
LIMIT 10;



(3)活跃用户统计（dws_user_active）
DROP TABLE IF EXISTS dws_user_active;

CREATE TABLE dws_user_active(
    user_id INT ,
    behavior_count INT,
    buy_count INT
);

INSERT INTO dws_user_active
SELECT
    user_id,
    COUNT(*) AS behavior_count,
    SUM(CASE WHEN behavior='buy' THEN 1 ELSE 0 END)
FROM fact_behavior
GROUP BY user_id;

SELECT *
FROM dws_user_active
ORDER BY behavior_count DESC
LIMIT 10;



(4)商品类别统计（dws_category_sales）
DROP TABLE IF EXISTS dws_category_sales;

CREATE TABLE dws_category_sales(
    category_id INT ,
    category VARCHAR(100),
    buy_count INT,
    sales_amount DECIMAL(12,2)

);

INSERT INTO dws_category_sales
SELECT
    c.category_id,
    c.category,
    COUNT(*) AS buy_count,
    ROUND(SUM(f.price),2) AS sales_amount
FROM fact_behavior f
JOIN dim_goods g
ON f.item_id=g.item_id
JOIN dim_category c
ON g.category_id=c.category_id
WHERE f.behavior='buy'
GROUP BY c.category_id,c.category;


SELECT *
FROM dws_category_sales
ORDER BY category DESC
LIMIT 10;
